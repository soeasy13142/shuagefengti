/**
 * GF(2^8) multiply by 2 (xtime).
 * @param {number} x - byte value 0-255
 * @returns {number}
 */
function xtime(x) {
  return ((x << 1) ^ (x & 0x80 ? 0x1b : 0)) & 0xff;
}
 
/**
 * GF(2^8) general multiplication.
 * Uses xtime-based binary multiplication.
 * @param {number} a - byte value 0-255
 * @param {number} b - byte value 0-255
 * @returns {number}
 */
function gfMul(a, b) {
  let result = 0;
  let current = a;
  let bit = b;
  while (bit > 0) {
    if (bit & 1) {
      result ^= current;
    }
    current = xtime(current);
    bit >>= 1;
  }
  return result;
}
 
/**
 * SubWord: apply S-box to each byte of a 4-byte word.
 * @param {number[]} w - 4-element array
 * @returns {number[]} new 4-element array
 */
function subWord(w) {
  return [
    sBox[w[0]],
    sBox[w[1]],
    sBox[w[2]],
    sBox[w[3]]
  ];
}
 
/**
 * RotWord: rotate a 4-byte word left by one byte.
 * @param {number[]} w - 4-element array
 * @returns {number[]} new 4-element array
 */
function rotWord(w) {
  return [w[1], w[2], w[3], w[0]];
}
 
/**
 * Rcon: round constant for key expansion.
 * Rcon[i] = [x^{i-1}, 0x00, 0x00, 0x00] in GF(2^8).
 * @param {number} i - round index (1-based)
 * @returns {number[]} 4-element array
 */
function rcon(i) {
  let val = 1;
  for (let j = 1; j < i; j++) {
    val = xtime(val);
  }
  return [val, 0x00, 0x00, 0x00];
}
