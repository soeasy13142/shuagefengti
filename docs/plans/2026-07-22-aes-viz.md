**Interfaces:**

`utils/aes-core.js`:
- `sBox`: `Uint8Array(256)` — FIPS 197 Figure 7 S-box (forward)
- `invSBox`（可选，第一版只实现加密）:
- `xtime(x: number)`: `number` — GF(2^8) ×2
- `gfMul(a: number, b: number)`: `number` — GF(2^8) 通用乘法
- `subWord(w: number[])`: `number[]` — 4 字节 S-box 替换
- `rotWord(w: number[])`: `number[]` — 4 字节循环左移 1 字节
- `rcon(i: number)`: `number[]` — 轮常量（Rcon[i] 0x00...）

`utils/aes-state.js`:
- `bytesToState(bytes: number[])`: `State` — 16 字节转 4×4 矩阵（列优先）
- `stateToBytes(state: State)`: `number[]` — 4×4 矩阵转 16 字节
- `subBytes(state: State)`: `State` — S-box 逐字节替换
- `shiftRows(state: State)`: `State` — 行移位
- `mixColumns(state: State)`: `State` — 列混合
- `addRoundKey(state: State, roundKey: number[])`: `State`

`utils/aes-key-expansion.js`:
- `keyExpansion(key: number[])`: `number[][]` — 返回 W[0..43]（44 words）
- `getRoundKey(w: number[][], round: number)`: `number[]` — 第 round 轮的轮密钥（16 字节）
- `keyExpansionWithSteps(key: number[])`: `{ words: number[][], steps: Array<{ index, type, input, output, rcon? }> }`

**Test vectors (must pass):**

FIPS 197 Appendix B:
- plaintext: `00112233445566778899aabbccddeeff`（16 字节）
- key: `000102030405060708090a0b0c0d0e0f`（16 字节）
- ciphertext: `69c4e0d86a7b0430d8cdb78070b4c55a`

FIPS 197 Appendix C 中间状态（每轮 start of round）：

```
Round 1 start:
  00102030405060708090a0b0c0d0e0f0  (after AddRoundKey with K0)
Round 2 start:
  89d810e8855ace682d1843d8cb128fe4
...
Round 10 start:
  ead27321b58dbad2312bf5607f8d292f  (before SubBytes of last round)
Round 10 output:
  69c4e0d86a7b0430d8cdb78070b4c55a  (ciphertext)
```

    const out = subBytes(input);
    // S-box of 0x00..0x0f
    const expectedBytes = [0x63, 0x7c, 0x77, 0x7b, 0xf2, 0x6b, 0x6f, 0xc5, 0x30, 0x01, 0x67, 0x2b, 0xfe, 0xd7, 0xab, 0x76];

  });
});

describe('shiftRows', () => {
  test('shifts rows correctly for known state', () => {
    // Row 0: unchanged, Row 1: left-1, Row 2: left-2, Row 3: left-3
    const input = bytesToState([0x00, 0x01, 0x02, 0x03, 0x10, 0x11, 0x12, 0x13, 0x20, 0x21, 0x22, 0x23, 0x30, 0x31, 0x32, 0x33]);
    const out = shiftRows(input);
    // After shiftRows: row0 unchanged, row1 [0x11, 0x12, 0x13, 0x10], row2 [0x22, 0x23, 0x20, 0x21], row3 [0x33, 0x30, 0x31, 0x32]

    expect(actual[0]).toBe(0x00);  // row0 col0
    expect(actual[5]).toBe(0x12);  // row1 col1 (shifted)
    expect(actual[10]).toBe(0x20); // row2 col2 (shifted)
    expect(actual[15]).toBe(0x32); // row3 col3 (shifted)
  });
});

describe('mixColumns', () => {
  test('FIPS 197 example column mixing', () => {
    // After SubBytes+ShiftRows of Round 1 in FIPS 197 Appendix C
    // Input column: d4 bf 5d 30 → Output column: 04 66 81 e5
    const input = bytesToState([
      0xd4, 0xe0, 0xb8, 0x1e, // actually this is the full state after ShiftRows round 1
      0xbf, 0xb4, 0x41, 0x27,
      0x5d, 0x52, 0x11, 0x98,
      0x30, 0xae, 0xf1, 0xe5
    ]);
    const out = mixColumns(input);
    const actual = stateToBytes(out);
    // First column: (02·d4)⊕(03·bf)⊕5d⊕30 = 04
    // FIPS 197 Appendix C Round 1 after MixColumns: first byte should be 0x04
    expect(actual[0]).toBe(0x04);
  });
});

describe('addRoundKey', () => {
  test('XOR state with round key', () => {
    const state = bytesToState(Array(16).fill(0xff));
    const key = Array(16).fill(0x01);
    const out = addRoundKey(state, key);
    const actual = stateToBytes(out);
    expect(actual.every(b => b === 0xfe)).toBe(true);
  });
});

    const key = hexToBytes('000102030405060708090a0b0c0d0e0f');
    const w = keyExpansion(key);
    expect(w[0]).toEqual([0x00, 0x01, 0x02, 0x03]);
    expect(w[1]).toEqual([0x04, 0x05, 0x06, 0x07]);
    expect(w[2]).toEqual([0x08, 0x09, 0x0a, 0x0b]);
    expect(w[3]).toEqual([0x0c, 0x0d, 0x0e, 0x0f]);
    expect(w[4]).toEqual([0xd6, 0xaa, 0x74, 0xfd]);
    expect(w[5]).toEqual([0xd2, 0xaf, 0x72, 0xfa]);
    expect(w[6]).toEqual([0xda, 0xa6, 0x78, 0xf1]);
    expect(w[7]).toEqual([0xd6, 0xab, 0x76, 0xfe]);
  });
  test('W[40..43] (last round key)', () => {
    const key = hexToBytes('000102030405060708090a0b0c0d0e0f');
    const w = keyExpansion(key);
    expect(w[40]).toEqual([0xb6, 0x63, 0x0c, 0xa6]);
  });
  test('returns 44 words', () => {
    const key = hexToBytes('000102030405060708090a0b0c0d0e0f');
    const w = keyExpansion(key);
    expect(w.length).toBe(44);
  });
});

describe('keyExpansionWithSteps', () => {
  test('returns steps including type and rcon', () => {
    const key = hexToBytes('000102030405060708090a0b0c0d0e0f');
    const result = keyExpansionWithSteps(key);
    expect(result.words.length).toBe(44);
    expect(result.steps.length).toBe(44);
    // W[4] should have type 'rotSubRcon' and rcon
    expect(result.steps[4].type).toBe('rotSubRcon');
    expect(result.steps[4].rcon).toBeDefined();
    // W[5] should have type 'direct'
    expect(result.steps[5].type).toBe('direct');
  });
});
